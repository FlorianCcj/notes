Tout dans un fichier¶

  • g++ -Wall -c main.cpp -o main.o && g++ main.o -o test && ./test

  • g++ main.cpp -o test && ./test

#include <iostream>

using namespace std;

class add {
  public:

    void sum(int int1, int int2)
    {
      _result = int1 + int2;
    }

    int getResult()
    {
      return result();
    }

    int result()
    {
      return _result;
    }

  private:
    int _result;
};

int main()
{
    add first_add;
    first_add.sum(1,2);
    cout << first_add.result() << endl;
    cout << "coucou" << endl;
    return 0;
}
